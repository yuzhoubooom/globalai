        <!-- 查询结果将显示在这里 -->

<script>

    const apiKeyInput = document.getElementById('apiKeyInput');

    const queryButton = document.getElementById('queryButton');

    const resultDiv = document.getElementById('result');


    queryButton.addEventListener('click', async () => {

        const apiKey = apiKeyInput.value.trim();


        if (!apiKey || !apiKey.startsWith('sk-')) {

            resultDiv.className = 'error';

            resultDiv.innerHTML = '请输入有效的 API Key (必须以 sk- 开头)';

            return;

        }


        resultDiv.className = 'loading';

        resultDiv.innerHTML = '<div class="loader"></div>';

        queryButton.disabled = true;


        try {

            // 注意！这里的 URL 变了！

            const response = await fetch('/api/proxy', {

                method: 'POST', // 请求方法变为POST

                headers: {

                    'Content-Type': 'application/json',

                },

                body: JSON.stringify({ apiKey: apiKey }), // 将API Key放在请求体中发送

            });


            const data = await response.json();


            if (!response.ok) {

                // 直接使用代理返回的错误信息

                throw new Error(data.message || `服务器返回错误，状态码: ${response.status}`);

            }

            

            if (data.success) {

                const quota = data.data.quota;

                const usd_balance = quota / 500000;

                

                resultDiv.className = 'success';

                resultDiv.innerHTML = `查询成功！<br><strong>剩余额度：$${usd_balance.toFixed(5)}</strong>`;

            } else {

                throw new Error(data.message || '查询失败，但未返回明确错误信息。');

            }


        } catch (error) {

            resultDiv.className = 'error';

            resultDiv.innerHTML = error.message;

        } finally {

            queryButton.disabled = false;

        }

    });

</script>
